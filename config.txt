# 图片画廊配置文件示例
# 复制此文件并重命名为 config.txt 以使用自定义配置

# 每行显示的图片数量（2-6之间推荐）
images_per_row=4

# 图片大小，单位为像素（80-200之间推荐）
image_size=100

# 常用配置示例：
# 紧凑布局：images_per_row=6, image_size=80
# 标准布局：images_per_row=4, image_size=100
# 宽松布局：images_per_row=2, image_size=150

# 注意事项：
# 1. 每行图片数量不宜过多，否则在小屏幕设备上可能显示不全
# 2. 图片大小过小会影响查看效果，过大会导致页面过长
# 3. 修改配置后，需要提交到GitHub仓库才能生效
